### encrypt.c

```c
/**
 * 设计思路
 * 1. 读取文件内容
 * 2. 将文件内容加密
 * 3. 使用加密后的文件内容创建一个新的临时文件
 * 4. 删除原来的文件
 * 5. 将临时文件改为原文件
 */

// 引入头文件用的，标准的输入输出
#include <stdio.h>
// 引入字符串， 它提供了一些字符串所要用到的函数
#include <string.h>
// 系统函数库
#include <stdlib.h>
#include <ctype.h>


/** 声明变量 */

// 保存文件的路径
char filename[256] = "";
// 宏指针，初始化为 NULL
FILE *fp = NULL;
// 临时保存，加密后的文件内容
FILE *fptemp = NULL;

// 存放密码
char password[256] = "";
// 密码长度
int passlen = 0;

// 创建临时文件
const char tempfile[256] = "tempfile.temp";

// 循环使用的计数器
int i = 0;
// 循环取文件内容时，存放每一个字符
char ch = '0';


/** 函数显示声明 */
void menu();
void inputpass(char *pass);


/** 函数实现 */
void menu()
{
    printf("***********************\n");
    printf("********安全课程********\n");
    printf("***********************\n");
    printf("**请输入要加密或解密的文件的路径**\n");
    // 1. 打开文件
    // 1.1 通过字符终端读取一个字符串，存放到 fliename 属性中
    // 因为gets()写法已经被废弃，下面的写法相当于 gets(filename);
    fgets(filename,255,stdin);
    filename[strcspn(filename,"\n")] = 0;

    // 1.2 通过文件指针来打开一个文件，所以文件指针也要先声明，它是通过宏来实现的
    //     在C语言中宏定义都是大写的 FILE *fp = NULL;
    if ( NULL == (fp = fopen(filename, "rb")))
    {
        printf("**输入的文件不存在**\n");
        // 0 正常退出
        // 正数 人为控制退出
        // 负数 异常退出
        exit(1);
    }
    // 1.3 文件存在
    printf("**请输入密码**\n");
    // 1.4 将控制台输入的密码保存到变量
    inputpass(password);
    passlen = (int)strlen(password);
    if(0 == passlen)
    {
        printf("**密码长度不能为0**\n");
        exit(2);
    }

    // 2. 读取文件内容，进行加密
    fptemp = fopen(tempfile, "wb");

    // 3. 写回加密文件
    // 3.1 死循环读取文件内容
    while (1)
    {
        ch = getc(fp);
        if (feof(fp))
        {
            /* 判断文件是否读完 */
            break;
        }
        // 每取出一个字符就加密
        ch ^= password[i++];
        // 将加密后的字符写入到临时文件
        fputc(ch, fptemp);
        //
        if (i == passlen)
        {
            i = 0;
        }

    }
    // 循环后清理
    // 关闭打开的文件
    fclose(fp);
    fclose(fptemp);
    // 把原文件删除
    remove(filename);
    // 把临时文件改原文件
    rename(tempfile,filename);
    printf("**文件加密或解密成功**\n");

}


void inputpass(char *pass)
{
    // 接收控制台输入的密码，并将密码保存到变量
    scanf("%s", pass);
}


// 主函数 是C语言程序的入口地址
int main(int argc, char const *argv[])
{
    menu();
    return 0;
}
```
