# SeekDB 分词器效果测试



``` sql
-- =====================================================
-- 统一测试文本（语义固定，不变）
-- Apple iPhone 是苹果最新发布的智能手机，用于日常通信和娱乐
-- =====================================================

-- SPACE 默认分词
select tokenize("Apple iPhone 是苹果最新发布的智能手机，用于日常通信和娱乐", 'space');

-- IK 分词（smart）
select tokenize("Apple iPhone 是苹果最新发布的智能手机，用于日常通信和娱乐",'IK','[{"additional_args":[{"ik_mode":"smart"}]}]');

-- IK 分词（max_word）
select tokenize("Apple iPhone 是苹果最新发布的智能手机，用于日常通信和娱乐",'IK','[{"additional_args":[{"ik_mode":"max_word"}]}]');

-- NGRAM 分词（2-gram）
select tokenize("Apple iPhone 是苹果最新发布的智能手机，用于日常通信和娱乐",'NGRAM','[{"additional_args":[{"ngram_token_size":2}]}]');

-- NGRAM 分词（3-gram）
select tokenize("Apple iPhone 是苹果最新发布的智能手机，用于日常通信和娱乐",'NGRAM','[{"additional_args":[{"ngram_token_size":3}]}]');

-- BENG 英文分词
select tokenize("Apple iPhone 是苹果最新发布的智能手机，用于日常通信和娱乐",'BENG','[]');

-- =====================================================
-- 同一语义（日文占主）
-- Apple iPhone は苹果が最新发布したスマートフォン、日常的な通信と娱乐に使われます
-- =====================================================

-- SPACE 默认分词
select tokenize("Apple iPhone は苹果が最新发布したスマートフォン、日常的な通信と娱乐に使われます", 'space');

-- IK（smart）
select tokenize("Apple iPhone は苹果が最新发布したスマートフォン、日常的な通信と娱乐に使われます",'IK','[{"additional_args":[{"ik_mode":"smart"}]}]');

-- IK 分词（max_word）
select tokenize("Apple iPhone は苹果が最新发布したスマートフォン、日常的な通信と娱乐に使われます",'IK','[{"additional_args":[{"ik_mode":"max_word"}]}]');

-- NGRAM 2-gram
select tokenize("Apple iPhone は苹果が最新发布したスマートフォン、日常的な通信と娱乐に使われます",'NGRAM','[{"additional_args":[{"ngram_token_size":2}]}]');

-- BENG
select tokenize("Apple iPhone は苹果が最新发布したスマートフォン、日常的な通信と娱乐に使われます",'BENG','[]');

-- =====================================================
-- 同一语义（英文占主）
-- Apple iPhone is Apple latest smartphone for communication and entertainment
-- =====================================================

-- SPACE 默认分词
select tokenize("Apple iPhone is Apple 最新发布的 smartphone for communication and entertainment", 'space');

-- IK（smart）
select tokenize("Apple iPhone is Apple 最新发布的 smartphone for communication and entertainment",'IK','[{"additional_args":[{"ik_mode":"smart"}]}]');

-- IK 分词（max_word）
select tokenize("Apple iPhone is Apple 最新发布的 smartphone for communication and entertainment",'IK','[{"additional_args":[{"ik_mode":"max_word"}]}]');

-- NGRAM 2-gram
select tokenize("Apple iPhone is Apple 最新发布的 smartphone for communication and entertainment",'NGRAM','[{"additional_args":[{"ngram_token_size":2}]}]');

-- BENG
select tokenize("Apple iPhone is Apple 最新发布的 smartphone for communication and entertainment",'BENG','[]');

```
