- S  服务器 10.249.87.4      /data/prod-sync/
- A  服务器 10.249.87.7     /data/prod-sync/
- B  服务器 110.249.87.8   /data/prod-sync/




#### 同步设计

``` shell
            10.249.87.4(源)
                /   \
               /     \
              /       \
             /         \
            /           \
           /             \
          v               v
     10.249.87.7       10.249.87.8
    
```

#### 在 10.249.87.4 开发服务器上创建`lsyncd.conf.lua`文件
``` lua
settings {
    logfile = "/var/log/lsyncd.log",
    statusFile = "/var/log/lsyncd.status",
    statusInterval = 10,   -- 每10秒更新状态
}

-- 同步到 10.249.87.7
sync {
    default.rsync,
    source = "/data/prod-sync/",
    target = "cloud@10.249.87.7:/data/prod-sync/",
    delay = 6,
    rsync = {
        archive = true,
        compress = true,
        _extra = {"--partial","delete"},
    }
}

-- 同步到 10.249.87.8
sync {
    default.rsync,
    source = "/data/prod-sync/",
    target = "cloud@10.249.87.8:/data/prod-sync/",
    delay = 6,
    rsync = {
        archive = true,
        compress = true,
        _extra = {"--partial","delete"},
    }
}

```

#### 使用

``` bash
ssh-keygen -t rsa -N "" -f ~/.ssh/id_rsa

ssh-copy-id -o StrictHostKeyChecking=no cloud@10.249.87.7
ssh-copy-id -o StrictHostKeyChecking=no cloud@10.249.87.8

```


``` bash
# 创建文件同步目录
mkdir -p /data/prod-sync/

# 创建工具存放配置文件的目录
mkdir -p /data/lsyncd/

cat > /data/lsyncd/lsyncd.conf.lua << ERIC

settings {
    logfile = "/var/log/lsyncd.log",
    statusFile = "/var/log/lsyncd.status",
    statusInterval = 10,   -- 每10秒更新状态
}

-- 同步到 10.249.87.7
sync {
    default.rsync,
    source = "/data/prod-sync/",
    target = "cloud@10.249.87.7:/data/prod-sync/",
    delay = 6,
    rsync = {
        archive = true,
        compress = true,
        _extra = {"--partial", "delete"},
    }
}

-- 同步到 10.249.87.8
sync {
    default.rsync,
    source = "/data/prod-sync/",
    target = "cloud@10.249.87.8:/data/prod-sync/",
    delay = 6,
    rsync = {
        archive = true,
        compress = true,
        _extra = {"--partial", "delete"},
    }
}

ERIC

```

``` bash
# 安装
apt install lsyncd

# 启动
lsyncd /data/lsyncd/lsyncd.conf.lua

# 自启动
systemctl enable lsyncd
```
