# RustFS 服务部署指南

本项目包含使用 Docker Compose 部署 RustFS 分布式存储服务的完整配置、环境变量以及初始化脚本。



``` bash
# 1. 创建项目专属根目录并进入
mkdir -p ./rustfs-service && cd ./rustfs-service

# 2. 创建宿主机挂载数据目录并配置权限
mkdir -p ./deploy/data/pro ./deploy/logs ./deploy/data/certs
chmod -R 777 ./deploy

# 3. 写入 docker-compose.yaml 文件
cat << 'EOF' > docker-compose.yaml
services:
  rustfs:
    security_opt:
      - "no-new-privileges:true"
    env_file:
      - ./rustfs.env
    image: rustfs/rustfs:latest
    container_name: rustfs-server
    ports:
      - "9000:9000" # S3 API 端口
      - "9001:9001" # Web 控制台端口
    volumes:
      - ./deploy/data/pro:/data
      - ./deploy/logs:/app/logs
      - ./deploy/data/certs/:/opt/tls
    networks:
      - rustfs-network
    restart: unless-stopped
    healthcheck:
      test:
        [
          "CMD",
          "sh", "-c",
          "curl -f http://127.0.0.1:9000/health && curl -f http://127.0.0.1:9001/rustfs/console/health"
        ]
      interval: 30s
      timeout: 10s
      retries: 3
      start_period: 40s

networks:
  rustfs-network:
    driver: bridge
EOF

# 4. 写入 rustfs.env 配置文件
cat << 'EOF' > rustfs.env
# ====== 凭证配置（部署前请修改为自定义复杂字符串） ======
RUSTFS_ACCESS_KEY=admin_access_key_xyz789
RUSTFS_SECRET_KEY=admin_secret_key_pro456file

# ====== 存储卷配置 ======
RUSTFS_VOLUMES=/data/rustfs{0..3}
# 允许在单块物理硬盘上跑多个卷（专为本地测试设计）
RUSTFS_UNSAFE_BYPASS_DISK_CHECK=true

# ====== 网络与服务监听 ======
RUSTFS_ADDRESS=0.0.0.0:9000
RUSTFS_CONSOLE_ENABLE=true
RUSTFS_CONSOLE_ADDRESS=0.0.0.0:9001
RUSTFS_CONSOLE_CORS_ALLOWED_ORIGINS=*

# ====== 证书与可观测性 ======
#RUSTFS_TLS_PATH=/opt/tls
RUSTFS_OBS_LOGGER_LEVEL=info
RUSTFS_OBS_ENDPOINT=http://otel-collector:4318
EOF

# 5. 启动服务
docker compose up -d
```
