## 编写测试程序`gpu_stress_test.py`

``` python
#!/usr/bin/env python3
"""
NVIDIA GPU 硬件故障检测脚本  v3.0

检测目标：
  1. 显存损坏  — 全显存写入读出校验 + ECC 错误计数
  2. 核心降频  — FP32 算力是否明显低于理论值
  3. 散热失效  — 满载温度 + 降频事件监测

用法:
    (base) root@Agent-Platform:~/test-gpu# uv run main.py
    ==============================================================
      NVIDIA GPU 硬件故障检测  v3.0
      2026-04-07 14:44:37  |  检测到 2 张 GPU
    ==============================================================
      [0]  NVIDIA A100 80GB PCIe              81920 MB
      [1]  NVIDIA A100 80GB PCIe              81920 MB
    --------------------------------------------------------------
    
      请选择要测试的 GPU（多选逗号或空格分隔，all = 全部）:
    
      输入编号 [0-1] 或 all: 0 1
    
      散热测试时长（直接回车 = 默认 10 分钟）:
        常用: 120 = 2 分钟  |  600 = 10 分钟  |  1800 = 30 分钟
      输入秒数 [默认 600]:

依赖:
    uv add torch numpy
"""

import subprocess, time, sys, json, argparse, os
from datetime import datetime
from multiprocessing import Process, Queue

try:
    import torch
    import numpy as np
except ImportError:
    print("[错误] 缺少依赖: uv add torch numpy")
    sys.exit(1)

if not torch.cuda.is_available():
    print("[错误] 未检测到 CUDA 设备，请确认驱动已正确安装。")
    sys.exit(1)


# ══════════════════════════════════════════════════════════════════════════════
# 工具
# ══════════════════════════════════════════════════════════════════════════════

def nvidia_smi(fields: list, gpu_idx: int) -> dict:
    cmd = ["nvidia-smi", f"--id={gpu_idx}",
           f"--query-gpu={','.join(fields)}", "--format=csv,noheader,nounits"]
    try:
        out = subprocess.check_output(cmd, text=True).strip().split("\n")[0]
        return dict(zip(fields, [v.strip() for v in out.split(",")]))
    except Exception:
        return {f: "N/A" for f in fields}


def list_gpus() -> list:
    cmd = ["nvidia-smi",
           "--query-gpu=index,name,memory.total,driver_version",
           "--format=csv,noheader,nounits"]
    try:
        lines = subprocess.check_output(cmd, text=True).strip().split("\n")
        result = []
        for line in lines:
            p = [x.strip() for x in line.split(",")]
            if len(p) >= 4:
                result.append({"index": int(p[0]), "name": p[1],
                                "vram_mb": p[2], "driver": p[3]})
        return result
    except Exception:
        return [{"index": i, "name": torch.cuda.get_device_name(i),
                 "vram_mb": "?", "driver": "?"}
                for i in range(torch.cuda.device_count())]


def parse_args(gpus: list):
    parser = argparse.ArgumentParser(description="NVIDIA GPU 硬件故障检测")
    parser.add_argument("--gpu",      type=int, default=None, help="测试单张卡，如 --gpu 0")
    parser.add_argument("--gpus",     type=str, default=None, help="测试多张卡，如 --gpus 0,1,2")
    parser.add_argument("--all",      action="store_true",    help="测试全部卡")
    parser.add_argument("--duration", type=int, default=None, metavar="秒",
                        help="散热测试时长（秒），默认交互询问")
    args = parser.parse_args()
    total = len(gpus)

    # 确定要测哪些卡
    if args.all:
        indices = list(range(total))
    elif args.gpus:
        indices = []
        for s in args.gpus.split(","):
            idx = int(s.strip())
            if not (0 <= idx < total):
                print(f"[错误] GPU {idx} 不存在（共 {total} 张）"); sys.exit(1)
            indices.append(idx)
    elif args.gpu is not None:
        if not (0 <= args.gpu < total):
            print(f"[错误] GPU {args.gpu} 不存在（共 {total} 张）"); sys.exit(1)
        indices = [args.gpu]
    else:
        # 交互选卡
        if total == 1:
            print(f"\n  只有 1 张卡，自动选择: [0] {gpus[0]['name']}\n")
            indices = [0]
        else:
            print("\n  请选择要测试的 GPU（多选逗号或空格分隔，all = 全部）:")
            while True:
                raw = input(f"\n  输入编号 [0-{total-1}] 或 all: ").strip().lower()
                if raw == "all":
                    indices = list(range(total)); break
                tokens = [t for t in raw.replace(",", " ").split() if t]
                try:
                    chosen = [int(t) for t in tokens]
                    if chosen and all(0 <= c < total for c in chosen):
                        indices = chosen; break
                    print(f"  [错误] 编号须在 0-{total-1} 范围内")
                except ValueError:
                    print("  [错误] 请输入数字")

    # 确定散热测试时长
    if args.duration is not None:
        duration = args.duration
    else:
        print("\n  散热测试时长（直接回车 = 默认 10 分钟）:")
        print("    常用: 120 = 2 分钟  |  600 = 10 分钟  |  1800 = 30 分钟")
        while True:
            raw = input("  输入秒数 [默认 600]: ").strip()
            if raw == "":
                duration = 600; break
            try:
                val = int(raw)
                if val > 0:
                    duration = val; break
                print("  [错误] 请输入正整数")
            except ValueError:
                print("  [错误] 请输入数字")

    return indices, duration


# FP32 理论值（CUDA Core，有据可查的型号）
THEORETICAL_FP32 = {
    # 数据中心
    "B200": 75.0, "B100": 60.0,
    "H100 SXM": 66.9, "H100 PCIe": 51.2, "H100": 51.2,
    "A100": 19.5, "A40": 37.4, "A10": 31.2,
    "V100 SXM": 15.7, "V100": 14.1,
    # RTX 40 系
    "RTX 4090": 82.6, "RTX 4080 SUPER": 52.2, "RTX 4080": 48.7,
    "RTX 4070 Ti SUPER": 40.0, "RTX 4070 Ti": 40.0,
    "RTX 4070 SUPER": 35.5, "RTX 4070": 29.1,
    "RTX 4060 Ti": 22.1, "RTX 4060": 15.1,
    # RTX 30 系
    "RTX 3090 Ti": 40.0, "RTX 3090": 35.6,
    "RTX 3080 Ti": 34.1, "RTX 3080": 29.8,
    "RTX 3070 Ti": 21.7, "RTX 3070": 20.3,
    "RTX 3060 Ti": 16.2, "RTX 3060": 12.7,
}

# 温度阈值（数据中心卡更严，消费级卡允许更高）
TEMP_WARN = {
    "A100": 83, "H100": 83, "B200": 85, "B100": 85,
    "A40": 83, "A10": 83, "V100": 83,
}
TEMP_WARN_DEFAULT = 90  # 消费级卡默认阈值


def get_theory(name: str) -> float | None:
    nu = name.upper()
    for key, val in THEORETICAL_FP32.items():
        if key.upper() in nu:
            return val
    return None


def get_temp_warn(name: str) -> int:
    nu = name.upper()
    for key, val in TEMP_WARN.items():
        if key.upper() in nu:
            return val
    return TEMP_WARN_DEFAULT


# ══════════════════════════════════════════════════════════════════════════════
# Logger
# ══════════════════════════════════════════════════════════════════════════════

class Logger:
    def __init__(self, gpu_idx: int, logfile):
        self.prefix = f"[GPU{gpu_idx}] "
        self.f = logfile

    def __call__(self, msg: str = ""):
        line = (self.prefix + msg) if msg else ""
        print(line, flush=True)
        self.f.write(line + "\n")
        self.f.flush()

    def ok(self, m):   self(f"[OK] {m}")
    def warn(self, m): self(f"[!!] {m}")
    def fail(self, m): self(f"[XX] {m}")
    def info(self, m): self(f"[..] {m}")
    def hr(self):      self("-" * 52)
    def section(self, t): self.hr(); self(f"  {t}"); self.hr()


# ══════════════════════════════════════════════════════════════════════════════
# 单卡检测（子进程）
# ══════════════════════════════════════════════════════════════════════════════

def run_single_gpu(gpu_idx: int, thermal_duration: int, result_queue: Queue):
    os.makedirs("logs", exist_ok=True)
    ts_start = datetime.now().strftime("%Y%m%d_%H%M%S")
    log_path = f"logs/gpu{gpu_idx}_test_{ts_start}.log"

    with open(log_path, "w", encoding="utf-8") as lf:
        log = Logger(gpu_idx, lf)
        torch.cuda.set_device(gpu_idx)
        device = torch.device(f"cuda:{gpu_idx}")

        faults = []   # 收集所有故障
        passed = []   # 收集通过的项目

        # ── 基本信息 ────────────────────────────────────────────────────────
        log.section("GPU 基本信息")
        d = nvidia_smi([
            "name", "driver_version", "memory.total", "memory.used",
            "temperature.gpu", "power.draw", "power.limit",
            "clocks.current.graphics", "fan.speed", "pstate",
        ], gpu_idx)
        name      = d.get("name", "Unknown")
        temp_idle = d.get("temperature.gpu", "?")
        temp_warn = get_temp_warn(name)

        log(f"  型号     : {name}")
        log(f"  驱动     : {d.get('driver_version','?')}")
        log(f"  显存     : {d.get('memory.used','?')} / {d.get('memory.total','?')} MB")
        log(f"  空闲温度 : {temp_idle} °C")
        log(f"  功耗     : {d.get('power.draw','?')} W / {d.get('power.limit','?')} W")
        log(f"  GPU 频率 : {d.get('clocks.current.graphics','?')} MHz")
        log()

        # ── 检测 1: ECC 错误 ─────────────────────────────────────────────────
        log.section("检测 1/3: 显存 ECC 错误")
        ecc = nvidia_smi([
            "ecc.errors.uncorrected.volatile.total",
            "ecc.errors.corrected.volatile.total",
        ], gpu_idx)
        ecc_uncorr = ecc.get("ecc.errors.uncorrected.volatile.total", "N/A")
        ecc_corr   = ecc.get("ecc.errors.corrected.volatile.total", "N/A")

        if ecc_uncorr not in ("N/A", "0", "[N/A]"):
            log.fail(f"ECC 不可修复错误: {ecc_uncorr}  ← 显存硬件损坏")
            faults.append(f"ECC 不可修复错误 {ecc_uncorr} 次")
        elif ecc_corr not in ("N/A", "0", "[N/A]"):
            log.warn(f"ECC 可修复错误: {ecc_corr}  ← 显存存在软错误，建议关注")
            faults.append(f"ECC 可修复错误 {ecc_corr} 次")
        else:
            log.ok("ECC 错误计数为 0")
            passed.append("ECC 正常")

        # ── 检测 2: 显存数据校验 ─────────────────────────────────────────────
        log.section("检测 2/3: 显存数据校验")
        log.info("分配全部可用显存，写入后读出比对...")
        torch.cuda.empty_cache()
        free_mb   = torch.cuda.mem_get_info(device)[0] // 1024**2
        chunk_mb  = 256
        chunks    = int(free_mb * 0.90) // chunk_mb
        tensors, patterns = [], []
        mem_ok = None
        try:
            for i in range(chunks):
                ne  = chunk_mb * 1024**2 // 4
                pat = float(i % 256)
                tensors.append(torch.full((ne,), pat, dtype=torch.float32, device=device))
                patterns.append(pat)
            torch.cuda.synchronize(device)

            errors = sum(1 for t, p in zip(tensors, patterns)
                         if not torch.all(t == p).item())
            total_mb = chunks * chunk_mb
            if errors == 0:
                log.ok(f"数据校验通过  ({total_mb} MB，{chunks} 块，无错误)")
                passed.append(f"显存数据校验通过 ({total_mb} MB)")
                mem_ok = True
            else:
                log.fail(f"数据校验失败！{errors}/{chunks} 块数据异常  ← 显存损坏")
                faults.append(f"显存数据校验失败 ({errors}/{chunks} 块)")
                mem_ok = False
        except torch.cuda.OutOfMemoryError:
            log.warn("显存分配 OOM，跳过校验")
            mem_ok = None
        finally:
            for t in tensors:
                del t
            torch.cuda.empty_cache()

        # ── 检测 3: 核心算力 + 散热 ─────────────────────────────────────────
        log.section(f"检测 3/3: 核心算力 & 散热 ({thermal_duration//60} 分 {thermal_duration%60} 秒)" if thermal_duration % 60
                    else f"检测 3/3: 核心算力 & 散热 ({thermal_duration//60} 分钟)")

        # FP32 算力（关闭 TF32，测纯 CUDA Core）
        log.info("FP32 算力测试...")
        torch.backends.cuda.matmul.allow_tf32 = False
        sz = 4096
        a  = torch.randn(sz, sz, dtype=torch.float32, device=device)
        b  = torch.randn(sz, sz, dtype=torch.float32, device=device)
        for _ in range(3):
            torch.mm(a, b)
        torch.cuda.synchronize(device)
        runs = 20
        t0 = time.perf_counter()
        for _ in range(runs):
            torch.mm(a, b)
        torch.cuda.synchronize(device)
        tflops = 2 * sz**3 * runs / (time.perf_counter() - t0) / 1e12
        del a, b

        theory = get_theory(name)
        if theory:
            ratio = tflops / theory * 100
            if ratio < 80:
                log.fail(f"FP32 算力: {tflops:.2f} TFLOPS  (理论 {theory}, 达成 {ratio:.1f}%)  ← 核心可能故障/降频")
                faults.append(f"FP32 算力偏低 ({ratio:.1f}% 理论值)")
            else:
                log.ok(f"FP32 算力: {tflops:.2f} TFLOPS  (理论 {theory}, 达成 {ratio:.1f}%)")
                passed.append(f"FP32 算力正常 ({ratio:.1f}%)")
        else:
            log.ok(f"FP32 算力: {tflops:.2f} TFLOPS  (该型号无理论参考值，仅供记录)")
            passed.append(f"FP32 算力 {tflops:.2f} TFLOPS")

        # 满载散热测试
        log.info(f"满载散热测试，每 30 秒采样...")
        log()
        log(f"  {'时间':>5}    {'温度':>4}    {'功耗':>4}         {'频率':>2}        {'状态':>1}")
        log(f"  {'------':>8}  {'------':>8}  {'----------':>9}   {'----------':>8}  {'--------':>8}")

        sa = torch.randn(sz, sz, dtype=torch.float32, device=device)
        sb = torch.randn(sz, sz, dtype=torch.float32, device=device)
        temps, throttle_events = [], 0
        deadline    = time.time() + thermal_duration
        last_sample = time.time() - 30

        try:
            while time.time() < deadline:
                for _ in range(20):
                    torch.mm(sa, sb)
                torch.cuda.synchronize(device)

                now = time.time()
                if now - last_sample >= 30:
                    d1 = nvidia_smi(["temperature.gpu", "power.draw",
                                     "clocks.current.graphics"], gpu_idx)
                    d2 = nvidia_smi(["clocks.throttle_reasons.hw_slowdown",
                                     "clocks.throttle_reasons.sw_thermal_slowdown"], gpu_idx)
                    temp  = d1.get("temperature.gpu", "?")
                    power = d1.get("power.draw", "?")
                    clk   = d1.get("clocks.current.graphics", "?")
                    throttle = (
                        "Active" in d2.get("clocks.throttle_reasons.hw_slowdown", "") or
                        "Active" in d2.get("clocks.throttle_reasons.sw_thermal_slowdown", "")
                    )
                    if throttle:
                        throttle_events += 1
                    flag    = "!! 降频" if throttle else "   正常"
                    elapsed = int(thermal_duration - (deadline - now))
                    log(f"  {elapsed:>4}s  {temp:>8}°C  {power:>9}W  {clk:>7}MHz  {flag:>7}")
                    try:
                        temps.append(float(temp))
                    except Exception:
                        pass
                    last_sample = now
        except KeyboardInterrupt:
            log("[!] 散热测试提前终止")

        del sa, sb
        torch.cuda.empty_cache()
        log()

        # 散热判断
        if temps:
            max_temp = max(temps)
            if max_temp >= temp_warn:
                log.fail(f"满载温度 {max_temp:.0f}°C  ≥ {temp_warn}°C  ← 散热可能失效")
                faults.append(f"满载温度过高 ({max_temp:.0f}°C)")
            else:
                log.ok(f"满载温度 {max_temp:.0f}°C  (阈值 {temp_warn}°C)")
                passed.append(f"散热正常 ({max_temp:.0f}°C)")

        if throttle_events == 0:
            log.ok("无热降频事件")
            passed.append("无降频")
        elif throttle_events <= 2:
            log.warn(f"降频 {throttle_events} 次（偶发）")
            faults.append(f"轻微降频 ({throttle_events} 次)")
        else:
            log.fail(f"频繁降频 {throttle_events} 次  ← 散热失效")
            faults.append(f"频繁降频 ({throttle_events} 次)")

        # ── 结论 ────────────────────────────────────────────────────────────
        log.hr()
        if faults:
            verdict = "发现故障"
            log(f"\n  结论: [故障]  发现以下问题:")
            for f in faults:
                log(f"    ✗ {f}")
        else:
            verdict = "硬件正常"
            log(f"\n  结论: [正常]  未发现硬件故障")
            for p in passed:
                log(f"    ✓ {p}")
        log()

        # 保存报告
        ts = datetime.now().strftime("%Y%m%d_%H%M%S")
        json_path = f"logs/gpu{gpu_idx}_report_{ts}.json"
        report = {
            "timestamp": datetime.now().isoformat(),
            "gpu_index": gpu_idx,
            "gpu_name":  name,
            "verdict":   verdict,
            "faults":    faults,
            "passed":    passed,
            "fp32_tflops": tflops,
            "max_temp":    max(temps) if temps else None,
            "throttle_events": throttle_events,
            "mem_check_ok": mem_ok,
        }
        with open(json_path, "w", encoding="utf-8") as f:
            json.dump(report, f, ensure_ascii=False, indent=2)
        log(f"报告: {json_path}")
        log(f"日志: {log_path}")

    result_queue.put({
        "gpu_idx":  gpu_idx,
        "name":     name,
        "verdict":  verdict,
        "faults":   faults,
        "fp32_tflops": tflops,
        "max_temp": max(temps) if temps else None,
    })


# ══════════════════════════════════════════════════════════════════════════════
# 汇总（多卡）
# ══════════════════════════════════════════════════════════════════════════════

def print_summary(results: list):
    results.sort(key=lambda r: r["gpu_idx"])
    w = 66
    print("\n" + "=" * w)
    print("  全部 GPU 检测完成 — 汇总")
    print("=" * w)
    for r in results:
        status = "[故障]" if r["faults"] else "[正常]"
        name   = r["name"][:30]
        temp   = f"{r['max_temp']:.0f}°C" if r["max_temp"] else "N/A"
        fp32   = f"{r['fp32_tflops']:.1f}T"
        print(f"  GPU {r['gpu_idx']}  {name:<30}  {status}  {fp32:>7}  {temp:>5}")
        for f in r["faults"]:
            print(f"         ✗ {f}")
    print("=" * w + "\n")


# ══════════════════════════════════════════════════════════════════════════════
# 主程序
# ══════════════════════════════════════════════════════════════════════════════

def main():
    from multiprocessing import set_start_method
    set_start_method("spawn", force=True)

    gpus = list_gpus()

    print("=" * 62)
    print("  NVIDIA GPU 硬件故障检测  v3.0")
    print(f"  {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}  |  检测到 {len(gpus)} 张 GPU")
    print("=" * 62)
    for g in gpus:
        print(f"  [{g['index']}]  {g['name']:<32}  {g['vram_mb']:>6} MB")
    print("-" * 62)

    indices, thermal_duration = parse_args(gpus)

    mins = thermal_duration // 60
    secs = thermal_duration % 60
    dur_str = f"{mins} 分 {secs} 秒" if secs else f"{mins} 分钟"
    print(f"\n  测试卡: {indices}   散热测试: {dur_str}")
    if len(indices) > 1:
        print(f"  并行运行 {len(indices)} 个子进程，日志写入 logs/ 目录")
    print()

    result_queue = Queue()
    processes = []
    for idx in indices:
        p = Process(target=run_single_gpu,
                    args=(idx, thermal_duration, result_queue),
                    name=f"gpu{idx}-checker")
        p.start()
        processes.append(p)

    for p in processes:
        p.join()

    results = []
    while not result_queue.empty():
        results.append(result_queue.get())

    if len(indices) > 1:
        print_summary(results)
    else:
        r = results[0] if results else {}
        status = "故障" if r.get("faults") else "正常"
        print(f"\n  检测完成: [{status}]  详见 logs/ 目录\n")


if __name__ == "__main__":
    main()
```
